x4 = layers.Dense(64, activation='relu')(x)

legtyp = layers.Dense(3, name='legtyp_out')(x4)

x5 = layers.Dense(64, activation='relu')(x)

legcol = layers.Dense(11, name='legocol_out')(x5)
